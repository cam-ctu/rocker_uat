> rowCollapse_R <- function(x, idxs, ..., useNames = TRUE) {
+   res <- c(x[1, 1], x[2, 2], x[3, 3], x[4, 1], x[5, 2],
+            x[6, 3], x[7, 1], x[8, 2], x[9, 3])
+   # Preserve names attribute?
+   if (useNames) {
+     names <- rownames(x)
+     if (!is.null(names)) names(res) <- names
+   }
+   res
+ }
