{
    debug <- isTRUE(getOption("future.debug"))
    if (debug) {
        mdebugf_push("future(..., label = %s) ...", sQuoteLabel(label))
        mdebugf("lazy: %s", lazy)
        mdebugf("stdout: %s", stdout)
        mdebugf("conditions: [n=%d] %s", length(conditions), 
            commaq(conditions))
        on.exit(mdebugf_pop())
    }
    if (substitute) 
        expr <- substitute(expr)
    t_start <- Sys.time()
    onReference <- getOption("future.globals.onReference")
    if (is.null(onReference)) 
        onReference <- "ignore"
    if (!is.null(globals)) {
        gp <- getGlobalsAndPackages(expr, envir = envir, tweak = tweakExpression, 
            globals = globals, onReference = onReference, maxSize = +Inf)
        expr <- gp[["expr"]]
        globals <- gp[["globals"]]
        if (length(packages) > 0 || length(gp[["packages"]]) > 
            0) {
            packages <- c(gp[["packages"]], packages)
        }
        gp <- NULL
        attr(globals, "already-done") <- TRUE
    }
    future <- Future(expr, substitute = FALSE, envir = envir, 
        lazy = TRUE, seed = seed, globals = globals, packages = packages, 
        stdout = stdout, conditions = conditions, label = label, 
        onReference = onReference, ...)
    args_names <- names(list(...))
    for (name in c("earlySignal", "gc", "local")) {
        if (!is.null(future[[name]])) {
            if (is.element(name, args_names)) {
                deprecateArgument("future", name, future[[name]])
            }
        }
    }
    future[[".defaultLocal"]] <- !is.element("local", args_names)
    if (isTRUE(getOption("future.journal"))) {
        future <- makeFutureJournal(future, event = "create", 
            category = "overhead", start = t_start)
    }
    if (!lazy) {
        future <- run(future)
        future[["lazy"]] <- FALSE

    }

        on.exit(mdebugf_pop())
    }
    if (substitute) 
        expr <- substitute(expr)

    if (is.null(onReference))

    if (!is.null(globals)) {

        expr <- gp[["expr"]]
        globals <- gp[["globals"]]
        if (length(packages) > 0 || length(gp[["packages"]]) >

            packages <- c(gp[["packages"]], packages)
        }

    }

        stdout = stdout, conditions = conditions, label = label, 
        onReference = onReference, ...)

    for (name in c("earlySignal", "gc", "local")) {

            }
        }

    if (isTRUE(getOption("future.journal"))) {
        future <- makeFutureJournal(future, event = "create", 
            category = "overhead", start = t_start)
    }
    if (!lazy) {

        mdebugf("conditions: [n=%d] %s", length(conditions), 
            commaq(conditions))
        on.exit(mdebugf_pop())
    }
    if (substitute) 
        expr <- substitute(expr)
    t_start <- Sys.time()
    onReference <- getOption("future.globals.onReference")
    if (is.null(onReference)) 
        onReference <- "ignore"
    if (!is.null(globals)) {
        gp <- getGlobalsAndPackages(expr, envir = envir, tweak = tweakExpression,

        expr <- gp[["expr"]]
        globals <- gp[["globals"]]
        if (length(packages) > 0 || length(gp[["packages"]]) >

            packages <- c(gp[["packages"]], packages)
        }
        gp <- NULL
        attr(globals, "already-done") <- TRUE
    }
    future <- Future(expr, substitute = FALSE, envir = envir,

        stdout = stdout, conditions = conditions, label = label, 
        onReference = onReference, ...)
    args_names <- names(list(...))
    for (name in c("earlySignal", "gc", "local")) {
        if (!is.null(future[[name]])) {
            if (is.element(name, args_names)) {
                deprecateArgument("future", name, future[[name]])
            }
        }
    }
    future[[".defaultLocal"]] <- !is.element("local", args_names)
    if (isTRUE(getOption("future.journal"))) {
        future <- makeFutureJournal(future, event = "create", 
            category = "overhead", start = t_start)
    }
    if (!lazy) {
        future <- run(future)
        future[["lazy"]] <- FALSE

    }
    future
