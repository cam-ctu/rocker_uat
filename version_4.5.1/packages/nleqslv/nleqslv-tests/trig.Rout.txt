> # Trigonometric function
> trig <- function(x) {
+ 	n <- length(x)
+ 	y <- cos(x)
+ 	s <- sum(y)
+ 	y <- n - s + c(1:n) * (1-y) - sin(x)
+ 
+ 	y
+ }
> 
> trigjac <- function(x) {
+ 	n <- length(x)
+ 	J <- matrix(numeric(n*n),n,n)
+ 
+ 	for (p in 1:n) {
+ 		J[,p]  <- sin(x[p])
+ 		J[p,p] <- (p+1) * sin(x[p]) - cos(x[p])
+ 	}
+ 
+ 	J
+ }
