> rowCumsums_R <- function(x, ..., useNames = NA) {
+   suppressWarnings({
+     y <- t(apply(x, MARGIN = 1L, FUN = cumsum))
+   })
+   
+   # Preserve dimnames attribute?
+   dim(y) <- dim(x)
+   if (isTRUE(useNames)) {
+     dimnames <- dimnames(x)
+     if (!is.null(dimnames)) dimnames(y) <- dimnames      
+   }
+ 
+   y
+ }
