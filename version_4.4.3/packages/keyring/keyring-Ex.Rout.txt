> ##D 
> ##D ## This is non-interactive, assuming that that default keyring
> ##D ## is unlocked

> ##D                    password = "secret")
> ##D key_get("R-keyring-test-service", "donaldduck")

> ##D key_delete("R-keyring-test-service", "donaldduck")
> ##D 
> ##D ## This is interactive using backend_file
> ##D ## Set variables to be used in keyring
> ##D kr_name <- "my_keyring"
> ##D kr_service <- "my_database"
> ##D kr_username <- "my_username"
> ##D 
> ##D ## Create a keyring and add an entry using the variables above
> ##D kb <- keyring::backend_file$new()
> ##D ## Prompt for the keyring password, used to unlock keyring
> ##D kb$keyring_create(kr_name)
> ##D ## Prompt for the secret/password to be stored in the keyring
> ##D kb$set(kr_service, username=kr_username, keyring=kr_name)
> ##D # Lock the keyring
> ##D kb$keyring_lock(kr_name)
> ##D 
> ##D ## The keyring file is stored at ~/.config/r-keyring/ on Linux
> ##D
